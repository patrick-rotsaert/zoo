		lockfile/lockfile_impl_windows.hpp
		logging/ibackend.cpp
		logging/spdlog_backend.cpp
		logging/logging.cpp
		misc/demangle.cpp
		misc/uuid.cpp
		misc/quoted_c.cpp
		misc/lock_types.hpp
		version.cpp
		version.h.in
		config.h.in
		apilinktest.cpp
	PUBLIC_HEADERS
		api.h
		compat.h
		conversion/conversion.h
		lockfile/lockfile.h
		logging/ibackend.h
		logging/spdlog_backend.h
		logging/logging.h
		logging/log_level.h
		misc/formatters.hpp
		misc/synchronized_queue.hpp
		misc/always_false.hpp
		misc/demangle.h
		misc/demangled_type_name.hpp
		misc/is_optional.hpp

if(ZOO_BUILD_EXAMPLES)
	add_subdirectory(examples)
